  \begin{tabular}{@{}l@{~~$\rightarrow$~~}l@{~~~~}l@{~~~~}l@{}}

\node[stage, fill=llmbg, below=0.4cm of s2] (s3) {%
  \textbf{\sffamily\small Stage 3: Constraint-Injected Code Generation}
  {\sffamily\footnotesize (LLM --- Attempt 1)}\\[6pt]
  %
  \begin{minipage}[t]{7.2cm}
    \footnotesize\sffamily
    \textbf{System:} \textit{You are an expert OpenROAD Python
    API programmer. Generate a script for the OpenROAD shell.}\\[2pt]
    \textbf{Rules:} \texttt{design}/\texttt{tech} pre-available;
    acquire in chain order; instance methods only;
    flat code; null-check \texttt{find*()} calls;
    provide \texttt{[Diagnosis]} if deviating from RAG.\\[4pt]
    \textbf{User:}\\
    \texttt{Task: Find which net is connected}\\
    \texttt{to pin A of instance inv\_1}\\[2pt]
    \texttt{Chain: Design $\to$ dbBlock $\to$ dbInst}\\
    \texttt{~~~~~~~$\to$ dbITerm $\to$ dbNet}\\[2pt]
    \texttt{Edge APIs:}\\
    \texttt{~~design.getBlock() $\to$ dbBlock}\\
    \texttt{~~block.findInst("inv\_1") $\to$ dbInst}\\
    \texttt{~~inst.findITerm("A") $\to$ dbITerm}\\

    \sffamily\textcolor{failred}{$\uparrow$ Bug: \texttt{getITerm()} does not}\\

    \textcolor{failred}{Also: no None-check on \texttt{findInst}.}
  \end{minipage}
};

  \begin{tabular}{@{}l@{~~}p{8.5cm}@{~~}l@{}}

};
